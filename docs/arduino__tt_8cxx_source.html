104 void handle_led(async_pulse_led *led) {

143 void loop()
144 {
145  bool tap_btn = digitalRead(TAP_BUTTON);
146  bool rst_btn = digitalRead(RESET_BUTTON);
147 
148  // Check for tap
149  bool pressed = CHECK_BTN_PRESSED(tap_btn_prev, tap_btn);
150  if (pressed) {
151  tt_tap(tt); // Register tap
152  tstamp = micros(); // Tempo period starts here
153  start_led_pulse(led, LED_PULSE_LEN_MS); // Start LED pulse
154  Serial.println("Tempo: " + String(tt_bpm(tt)) + " BPM");
155  DEBOUNCE(); // Debounce the button
156  }
157 
158  // Check for reset
159  pressed = CHECK_BTN_PRESSED(rst_btn_prev, rst_btn);
160  if (pressed) {
161  tt_reset(tt); // Reset tempo tapper
162  cancel_led_pulse(led); // Abort any ongoing LED pulse
163  Serial.println("Reset!");
164  DEBOUNCE(); // Debounce the button
165  }
166 
167  // Pulse LED at the current tempo
168  unsigned long period = tt_period_us(tt);
169  if (period > 0 && micros() - tstamp >= period) {
170  tstamp = micros();
171  start_led_pulse(led, LED_PULSE_LEN_MS); // Start LED pulse
172  }
173 
174  handle_led(led); // Handle LED pulse
175  tap_btn_prev = tap_btn;
176  rst_btn_prev = rst_btn;
177 }
